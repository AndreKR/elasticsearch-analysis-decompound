016    package org.xbib.elasticsearch.plugin.analysis.decompound;
017    
018    import org.elasticsearch.index.analysis.AnalysisModule;
019    import org.xbib.elasticsearch.index.analysis.DecompoundTokenFilterFactory;
020    import org.xbib.elasticsearch.index.analysis.GermanNormalizationFilterFactory;
021    import org.elasticsearch.plugins.AbstractPlugin;
022    
023    public class AnalysisDecompoundPlugin extends AbstractPlugin {
024    
025        @Override public String name() {
026            return "analysis-decompound";
027        }
028    
029        @Override public String description() {
030            return "A decompounding token filter for german and other languages";
031        }
032    
033        public void onModule(AnalysisModule module) {
034            module.addTokenFilter("decompound", DecompoundTokenFilterFactory.class);
035            module.addTokenFilter("german_normalize", GermanNormalizationFilterFactory.class);
036        }
037    }
