016    package org.xbib.elasticsearch.index.analysis;
017    
018    import java.io.IOException;
019    import java.util.LinkedList;
020    import org.apache.lucene.analysis.TokenFilter;
021    import org.apache.lucene.analysis.TokenStream;
022    import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
023    import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
024    import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;
025    import org.apache.lucene.util.AttributeSource;
026    import org.xbib.elasticsearch.analysis.decompound.Decompounder;
027    
028    public class DecompoundTokenFilter extends TokenFilter {
029    
030        protected final LinkedList<DecompoundToken> tokens;
031        protected final Decompounder decomp;
032        protected final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
033        protected final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
034        private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
035        private AttributeSource.State current;
036    
037        protected DecompoundTokenFilter(TokenStream input, Decompounder decomp) {
038            super(input);
039            this.tokens = new LinkedList<DecompoundToken>();
040            this.decomp = decomp;
041        }
042    
043        @Override
044        public final boolean incrementToken() throws IOException {
045            if (!tokens.isEmpty()) {
046                assert current != null;
047                DecompoundToken token = tokens.removeFirst();
048                restoreState(current);
049                termAtt.setEmpty().append(token.txt);
050                offsetAtt.setOffset(token.startOffset, token.endOffset);
051                posIncAtt.setPositionIncrement(0);
052                return true;
053            }
054            if (input.incrementToken()) {
055                decompound();
056                if (!tokens.isEmpty()) {
057                    current = captureState();
058                }
059                return true;
060            } else {
061                return false;
062            }
063        }
064    
065        protected void decompound() {
066            int start = offsetAtt.startOffset();
067            CharSequence term = new String(termAtt.buffer(), 0, termAtt.length());
068            for (String s : decomp.decompound(term.toString())) {
069                int len = s.length();
070                tokens.add(new DecompoundToken(s, start, len));
071                start += len;
072            }
073        }
074    
075        @Override
076        public void reset() throws IOException {
077            super.reset();
078            tokens.clear();
079            current = null;
080        }
081    
082        protected class DecompoundToken {
083    
084            public final CharSequence txt;
085            public final int startOffset;
086            public final int endOffset;
087    
088            public DecompoundToken(CharSequence txt, int offset, int length) {
089                this.txt = txt;
090                int startOff = DecompoundTokenFilter.this.offsetAtt.startOffset();
091                int endOff = DecompoundTokenFilter.this.offsetAtt.endOffset();
092                if (endOff - startOff != DecompoundTokenFilter.this.termAtt.length()) {
093                    this.startOffset = startOff;
094                    this.endOffset = endOff;
095                } else {
096                    this.startOffset = offset;
097                    this.endOffset = offset + length;
098                }
099            }
100        }
101    }
