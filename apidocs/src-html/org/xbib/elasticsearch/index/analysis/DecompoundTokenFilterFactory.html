016    package org.xbib.elasticsearch.index.analysis;
017    
018    import java.io.IOException;
019    import org.apache.lucene.analysis.TokenStream;
020    import org.elasticsearch.ElasticsearchIllegalArgumentException;
021    import org.xbib.elasticsearch.analysis.decompound.Decompounder;
022    import org.elasticsearch.common.inject.Inject;
023    import org.elasticsearch.common.inject.assistedinject.Assisted;
024    import org.elasticsearch.common.settings.Settings;
025    import org.elasticsearch.env.Environment;
026    import org.elasticsearch.index.Index;
027    import org.elasticsearch.index.analysis.AbstractTokenFilterFactory;
028    import org.elasticsearch.index.settings.IndexSettings;
029    
030    public class DecompoundTokenFilterFactory extends AbstractTokenFilterFactory {
031    
032        private final Decompounder decompounder;
033    
034        @Inject
035        public DecompoundTokenFilterFactory(Index index,
036                @IndexSettings Settings indexSettings, Environment env,
037                @Assisted String name, @Assisted Settings settings) {
038            super(index, indexSettings, name, settings);
039            this.decompounder = createDecompounder(env, settings);
040        }
041    
042        @Override
043        public TokenStream create(TokenStream tokenStream) {
044            return new DecompoundTokenFilter(tokenStream, decompounder);
045        }
046    
047        private Decompounder createDecompounder(Environment env, Settings settings) {
048            try {
049                String forward = settings.get("forward", "/kompVVic.tree");
050                String backward = settings.get("backward", "/kompVHic.tree");
051                String reduce = settings.get("reduce", "/grfExt.tree");
052                double threshold = settings.getAsDouble("threshold", 0.51);
053                return new Decompounder(env.resolveConfig(forward).openStream(),
054                        env.resolveConfig(backward).openStream(),
055                        env.resolveConfig(reduce).openStream(),
056                        threshold);
057            } catch (ClassNotFoundException e) {
058                throw new ElasticsearchIllegalArgumentException("decompounder resources in settings not found: " + settings, e);
059            } catch (IOException e) {
060                throw new ElasticsearchIllegalArgumentException("decompounder resources in settings not found: " + settings, e);
061            }
062        }
063    }
