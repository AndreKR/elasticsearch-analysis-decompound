026package org.elasticsearch.tools.opt;
027
028import java.util.Collection;
029
030/**
031 * Allows callers to specify whether a given option accepts arguments (required or optional).
032 *
033 * <p>Instances are returned from {@link OptionParser#accepts(String)} to allow the formation of parser directives as
034 * sentences in a "fluent interface" language.  For example:</p>
035 *
036 * <pre><code>
037 *   OptionParser parser = new OptionParser();
038 *   parser.accepts( "c" ).<strong>withRequiredArg()</strong>.ofType( Integer.class );
039 * </code></pre>
040 *
041 * <p>If no methods are invoked on an instance of this class, then that instance's option will accept no argument.</p>
042 *
043 * <p>Note that you should not use the fluent interface clauses in a way that would defeat the typing of option
044 * arguments:</p>
045 *
046 * <pre><code>
047 *   OptionParser parser = new OptionParser();
048 *   ArgumentAcceptingOptionSpec&lt;String&gt; optionC =
049 *       parser.accepts( "c" ).withRequiredArg();
050 *   <strong>optionC.ofType( Integer.class );  // DON'T THROW AWAY THE TYPE!</strong>
051 *
052 *   String value = parser.parse( "-c", "2" ).valueOf( optionC );  // ClassCastException
053 * </code></pre>
054 *
055 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
056 */
057public class OptionSpecBuilder extends NoArgumentOptionSpec {
058    private final OptionParser parser;
059
060    OptionSpecBuilder( OptionParser parser, Collection<String> options, String description ) {
061        super( options, description );
062
063        this.parser = parser;
064        attachToParser();
065    }
066
067    private void attachToParser() {
068        parser.recognize( this );
069    }
070
071    /**
072     * Informs an option parser that this builder's option requires an argument.
073     *
074     * @return a specification for the option
075     */
076    public ArgumentAcceptingOptionSpec<String> withRequiredArg() {
077        ArgumentAcceptingOptionSpec<String> newSpec =
078            new RequiredArgumentOptionSpec<String>( options(), description() );
079        parser.recognize( newSpec );
080
081        return newSpec;
082    }
083
084    /**
085     * Informs an option parser that this builder's option accepts an optional argument.
086     *
087     * @return a specification for the option
088     */
089    public ArgumentAcceptingOptionSpec<String> withOptionalArg() {
090        ArgumentAcceptingOptionSpec<String> newSpec =
091            new OptionalArgumentOptionSpec<String>( options(), description() );
092        parser.recognize( newSpec );
093
094        return newSpec;
095    }
096}
