026package org.elasticsearch.tools.opt;
027
028import java.util.ArrayList;
029import java.util.HashMap;
030import java.util.IdentityHashMap;
031import java.util.List;
032import java.util.Map;
033
034import static java.util.Collections.*;
035
036import static org.elasticsearch.tools.opt.internal.Objects.*;
037
038/**
039 * Representation of a group of detected command line options, their arguments, and non-option arguments.
040 *
041 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
042 */
043public class OptionSet {
044    private final List<OptionSpec<?>> detectedSpecs;
045    private final Map<String, AbstractOptionSpec<?>> detectedOptions;
046    private final Map<AbstractOptionSpec<?>, List<String>> optionsToArguments;
047    private final List<String> nonOptionArguments;
048    private final Map<String, List<?>> defaultValues;
049
050    /*
051     * Package-private because clients don't create these.
052     */
053    OptionSet( Map<String, List<?>> defaults ) {
054        detectedSpecs = new ArrayList<OptionSpec<?>>();
055        detectedOptions = new HashMap<String, AbstractOptionSpec<?>>();
056        optionsToArguments = new IdentityHashMap<AbstractOptionSpec<?>, List<String>>();
057        nonOptionArguments = new ArrayList<String>();
058        defaultValues = new HashMap<String, List<?>>( defaults );
059    }
060
061    /**
062     * Tells whether any options were detected.
063     * 
064     * @return {@code true} if any options were detected
065     */
066    public boolean hasOptions() {
067        return !detectedOptions.isEmpty();
068    }
069    
070    /**
071     * Tells whether the given option was detected.
072     *
073     * @param option the option to search for
074     * @return {@code true} if the option was detected
075     * @see #has(OptionSpec)
076     */
077    public boolean has( String option ) {
078        return detectedOptions.containsKey( option );
079    }
080
081    /**
082     * Tells whether the given option was detected.
083     *
084     * <p>This method recognizes only instances of options returned from the fluent interface methods.</p>
085     *
086     * <p>Specifying a {@linkplain ArgumentAcceptingOptionSpec#defaultsTo(Object, Object[])} default argument value}
087     * for an option does not cause this method to return {@code true} if the option was not detected on the command
088     * line.</p>
089     *
090     * @param option the option to search for
091     * @return {@code true} if the option was detected
092     * @see #has(String)
093     */
094    public boolean has( OptionSpec<?> option ) {
095        return optionsToArguments.containsKey( option );
096    }
097
098    /**
099     * Tells whether there are any arguments associated with the given option.
100     *
101     * @param option the option to search for
102     * @return {@code true} if the option was detected and at least one argument was detected for the option
103     * @see #hasArgument(OptionSpec)
104     */
105    public boolean hasArgument( String option ) {
106        AbstractOptionSpec<?> spec = detectedOptions.get( option );
107        return spec != null && hasArgument( spec );
108    }
109
110    /**
111     * Tells whether there are any arguments associated with the given option.
112     *
113     * <p>This method recognizes only instances of options returned from the fluent interface methods.</p>
114     *
115     * <p>Specifying a {@linkplain ArgumentAcceptingOptionSpec#defaultsTo(Object, Object[]) default argument value}
116     * for an option does not cause this method to return {@code true} if the option was not detected on the command
117     * line, or if the option can take an optional argument but did not have one on the command line.</p>
118     *
119     * @param option the option to search for
120     * @return {@code true} if the option was detected and at least one argument was detected for the option
121     * @throws NullPointerException if {@code option} is {@code null}
122     * @see #hasArgument(String)
123     */
124    public boolean hasArgument( OptionSpec<?> option ) {
125        ensureNotNull( option );
126
127        List<String> values = optionsToArguments.get( option );
128        return values != null && !values.isEmpty();
129    }
130
131    /**
132     * Gives the argument associated with the given option.  If the option was given an argument type, the argument
133     * will take on that type; otherwise, it will be a {@link String}.
134     *
135     * <p>Specifying a {@linkplain ArgumentAcceptingOptionSpec#defaultsTo(Object, Object[]) default argument value}
136     * for an option will cause this method to return that default value even if the option was not detected on the
137     * command line, or if the option can take an optional argument but did not have one on the command line.</p>
138     *
139     * @param option the option to search for
140     * @return the argument of the given option; {@code null} if no argument is present, or that option was not
141     * detected
142     * @throws NullPointerException if {@code option} is {@code null}
143     * @throws OptionException if more than one argument was detected for the option
144     */
145    public Object valueOf( String option ) {
146        ensureNotNull( option );
147
148        AbstractOptionSpec<?> spec = detectedOptions.get( option );
149        if ( spec == null ) {
150            List<?> defaults = defaultValuesFor( option );
151            return defaults.isEmpty() ? null : defaults.get( 0 );
152        }
153
154        return valueOf( spec );
155    }
156
157    /**
158     * Gives the argument associated with the given option.
159     *
160     * <p>This method recognizes only instances of options returned from the fluent interface methods.</p>
161     *
162     * @param <V> represents the type of the arguments the given option accepts
163     * @param option the option to search for
164     * @return the argument of the given option; {@code null} if no argument is present, or that option was not
165     * detected
166     * @throws OptionException if more than one argument was detected for the option
167     * @throws NullPointerException if {@code option} is {@code null}
168     * @throws ClassCastException if the arguments of this option are not of the expected type
169     */
170    public <V> V valueOf( OptionSpec<V> option ) {
171        ensureNotNull( option );
172
173        List<V> values = valuesOf( option );
174        switch ( values.size() ) {
175            case 0:
176                return null;
177            case 1:
178                return values.get( 0 );
179            default:
180                throw new MultipleArgumentsForOptionException( option.options() );
181        }
182    }
183
184    /**
185     * <p>Gives any arguments associated with the given option.  If the option was given an argument type, the
186     * arguments will take on that type; otherwise, they will be {@link String}s.</p>
187     *
188     * @param option the option to search for
189     * @return the arguments associated with the option, as a list of objects of the type given to the arguments; an
190     * empty list if no such arguments are present, or if the option was not detected
191     * @throws NullPointerException if {@code option} is {@code null}
192     */
193    public List<?> valuesOf( String option ) {
194        ensureNotNull( option );
195
196        AbstractOptionSpec<?> spec = detectedOptions.get( option );
197        return spec == null ? defaultValuesFor( option ) : valuesOf( spec );
198    }
199
200    /**
201     * <p>Gives any arguments associated with the given option.  If the option was given an argument type, the
202     * arguments will take on that type; otherwise, they will be {@link String}s.</p>
203     *
204     * <p>This method recognizes only instances of options returned from the fluent interface methods.</p>
205     *
206     * @param <V> represents the type of the arguments the given option accepts
207     * @param option the option to search for
208     * @return the arguments associated with the option; an empty list if no such arguments are present, or if the
209     * option was not detected
210     * @throws NullPointerException if {@code option} is {@code null}
211     * @throws OptionException if there is a problem converting the option's arguments to the desired type; for
212     * example, if the type does not implement a correct conversion constructor or method
213     */
214    public <V> List<V> valuesOf( OptionSpec<V> option ) {
215        ensureNotNull( option );
216
217        List<String> values = optionsToArguments.get( option );
218        if ( values == null || values.isEmpty() )
219            return defaultValueFor( option );
220
221        AbstractOptionSpec<V> spec = (AbstractOptionSpec<V>) option;
222        List<V> convertedValues = new ArrayList<V>();
223        for ( String each : values )
224            convertedValues.add( spec.convert( each ) );
225
226        return unmodifiableList( convertedValues );
227    }
228
229    /**
230     * Gives the set of options that were detected, in the form of {@linkplain OptionSpec}s, in the order in which the
231     * options were found on the command line.
232     *
233     * @return the set of detected command line options
234     */
235    public List<OptionSpec<?>> specs() {
236        return unmodifiableList( detectedSpecs );
237    }
238
239    /**
240     * @return the detected non-option arguments
241     */
242    public List<String> nonOptionArguments() {
243        return unmodifiableList( nonOptionArguments );
244    }
245    
246    void add( AbstractOptionSpec<?> spec ) {
247        addWithArgument( spec, null );
248    }
249
250    void addWithArgument( AbstractOptionSpec<?> spec, String argument ) {
251        detectedSpecs.add( spec );
252
253        for ( String each : spec.options() )
254            detectedOptions.put( each, spec );
255
256        List<String> optionArguments = optionsToArguments.get( spec );
257
258        if ( optionArguments == null ) {
259            optionArguments = new ArrayList<String>();
260            optionsToArguments.put( spec, optionArguments );
261        }
262
263        if ( argument != null )
264            optionArguments.add( argument );
265    }
266
267    void addNonOptionArgument( String argument ) {
268        nonOptionArguments.add( argument );
269    }
270
271    @Override
272    public boolean equals( Object that ) {
273        if ( this == that )
274            return true;
275
276        if ( that == null || !getClass().equals( that.getClass() ) )
277            return false;
278
279        OptionSet other = (OptionSet) that;
280        Map<AbstractOptionSpec<?>, List<String>> thisOptionsToArguments =
281            new HashMap<AbstractOptionSpec<?>, List<String>>( optionsToArguments );
282        Map<AbstractOptionSpec<?>, List<String>> otherOptionsToArguments =
283            new HashMap<AbstractOptionSpec<?>, List<String>>( other.optionsToArguments );
284        return detectedOptions.equals( other.detectedOptions )
285            && thisOptionsToArguments.equals( otherOptionsToArguments )
286            && nonOptionArguments.equals( other.nonOptionArguments() );
287    }
288
289    @Override
290    public int hashCode() {
291        Map<AbstractOptionSpec<?>, List<String>> thisOptionsToArguments =
292            new HashMap<AbstractOptionSpec<?>, List<String>>( optionsToArguments );
293        return detectedOptions.hashCode()
294            ^ thisOptionsToArguments.hashCode()
295            ^ nonOptionArguments.hashCode();
296    }
297
298    private <V> List<V> defaultValuesFor( String option ) {
299        if ( defaultValues.containsKey( option ) )
300            return (List<V>) defaultValues.get( option );
301
302        return emptyList();
303    }
304
305    private <V> List<V> defaultValueFor( OptionSpec<V> option ) {
306        return defaultValuesFor( option.options().iterator().next() );
307    }
308}
