026package org.elasticsearch.tools.opt.internal;
027
028import java.lang.reflect.Constructor;
029import java.lang.reflect.InvocationTargetException;
030import java.lang.reflect.Method;
031import static java.lang.reflect.Modifier.*;
032
033import org.elasticsearch.tools.opt.ValueConverter;
034
035import static org.elasticsearch.tools.opt.internal.Classes.*;
036
037/**
038 * <p>Helper methods for reflection.</p>
039 *
040 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
041 */
042public final class Reflection {
043    private Reflection() {
044        throw new UnsupportedOperationException();
045    }
046
047    /**
048     * Finds an appropriate value converter for the given class.
049     *
050     * @param <V> a constraint on the class object to introspect
051     * @param clazz class to introspect on
052     * @return a converter method or constructor
053     */
054    public static <V> ValueConverter<V> findConverter( Class<V> clazz ) {
055        Class<V> maybeWrapper = wrapperOf( clazz );
056
057        ValueConverter<V> valueOf = valueOfConverter( maybeWrapper );
058        if ( valueOf != null )
059            return valueOf;
060
061        ValueConverter<V> constructor = constructorConverter( maybeWrapper );
062        if ( constructor != null )
063            return constructor;
064
065        throw new IllegalArgumentException( clazz + " is not a value type" );
066    }
067
068    private static <V> ValueConverter<V> valueOfConverter( Class<V> clazz ) {
069        try {
070            Method valueOf = clazz.getDeclaredMethod( "valueOf", String.class );
071            if ( meetsConverterRequirements( valueOf, clazz ) )
072                return new MethodInvokingValueConverter<V>( valueOf, clazz );
073
074            return null;
075        }
076        catch ( NoSuchMethodException ignored ) {
077            return null;
078        }
079    }
080
081    private static <V> ValueConverter<V> constructorConverter( Class<V> clazz ) {
082        try {
083            return new ConstructorInvokingValueConverter<V>( clazz.getConstructor( String.class ) );
084        }
085        catch ( NoSuchMethodException ignored ) {
086            return null;
087        }
088    }
089
090    /**
091     * Invokes the given constructor with the given arguments.
092     *
093     * @param <T> constraint on the type of the objects yielded by the constructor
094     * @param constructor constructor to invoke
095     * @param args arguments to hand to the constructor
096     * @return the result of invoking the constructor
097     * @throws ReflectionException in lieu of the gaggle of reflection-related exceptions
098     */
099    public static <T> T instantiate( Constructor<T> constructor, Object... args ) {
100        try {
101            return constructor.newInstance( args );
102        }
103        catch ( Exception ex ) {
104            throw reflectionException( ex );
105        }
106    }
107
108    /**
109     * Invokes the given static method with the given arguments.
110     *
111     * @param method method to invoke
112     * @param args arguments to hand to the method
113     * @return the result of invoking the method
114     * @throws ReflectionException in lieu of the gaggle of reflection-related exceptions
115     */
116    public static Object invoke( Method method, Object... args ) {
117        try {
118            return method.invoke( null, args );
119        }
120        catch ( Exception ex ) {
121            throw reflectionException( ex );
122        }
123    }
124
125    private static boolean meetsConverterRequirements( Method method, Class<?> expectedReturnType ) {
126        int modifiers = method.getModifiers();
127        return isPublic( modifiers ) && isStatic( modifiers ) && expectedReturnType.equals( method.getReturnType() );
128    }
129
130    private static RuntimeException reflectionException( Exception ex ) {
131        if ( ex instanceof IllegalArgumentException )
132            return new ReflectionException( ex );
133        if ( ex instanceof InvocationTargetException )
134            return new ReflectionException( ex.getCause() );
135        if ( ex instanceof RuntimeException )
136            return (RuntimeException) ex;
137
138        return new ReflectionException( ex );
139    }
140}
