026package org.elasticsearch.tools.opt.internal;
027
028import java.util.Iterator;
029import java.util.List;
030import static java.lang.System.*;
031import static java.util.Arrays.*;
032
033/**
034 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
035 */
036public final class Strings {
037    public static final String EMPTY = "";
038    public static final String SINGLE_QUOTE = "'";
039    public static final String LINE_SEPARATOR = getProperty( "line.separator" );
040
041    private Strings() {
042        throw new UnsupportedOperationException();
043    }
044
045    /**
046     * <p>Gives a string consisting of the given character repeated the given number of
047     * times.</p>
048     *
049     * @param ch the character to repeat
050     * @param count how many times to repeat the character
051     * @return the resultant string
052     */
053    public static String repeat( char ch, int count ) {
054        StringBuilder buffer = new StringBuilder();
055
056        for ( int i = 0; i < count; ++i )
057            buffer.append( ch );
058
059        return buffer.toString();
060    }
061
062    /**
063     * <p>Tells whether the given string is either {@code} or consists solely of
064     * whitespace characters.</p>
065     *
066     * @param target string to check
067     * @return {@code true} if the target string is null or empty
068     */
069    public static boolean isNullOrEmpty( String target ) {
070        return target == null || EMPTY.equals( target );
071    }
072
073
074    /**
075     * <p>Gives a string consisting of a given string prepended and appended with
076     * surrounding characters.</p>
077     *
078     * @param target a string
079     * @param begin character to prepend
080     * @param end character to append
081     * @return the surrounded string
082     */
083    public static String surround( String target, char begin, char end ) {
084        return begin + target + end;
085    }
086
087    /**
088     * Gives a string consisting of the elements of a given array of strings, each
089     * separated by a given separator string.
090     *
091     * @param pieces the strings to join
092     * @param separator the separator
093     * @return the joined string
094     */
095    public static String join( String[] pieces, String separator ) {
096        return join( asList( pieces ), separator );
097    }
098
099    /**
100     * Gives a string consisting of the string representations of the elements of a
101     * given array of objects, each separated by a given separator string.
102     *
103     * @param pieces the elements whose string representations are to be joined
104     * @param separator the separator
105     * @return the joined string
106     */
107    public static String join( List<String> pieces, String separator ) {
108        StringBuilder buffer = new StringBuilder();
109
110        for ( Iterator<String> iter = pieces.iterator(); iter.hasNext(); ) {
111            buffer.append( iter.next() );
112
113            if ( iter.hasNext() )
114                buffer.append( separator );
115        }
116
117        return buffer.toString();
118    }
119}
