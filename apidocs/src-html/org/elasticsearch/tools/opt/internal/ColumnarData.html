026package org.elasticsearch.tools.opt.internal;
027
028import java.util.Iterator;
029import java.util.LinkedList;
030import java.util.List;
031import static java.lang.Integer.*;
032import static java.lang.System.*;
033import static java.util.Collections.*;
034
035import static org.elasticsearch.tools.opt.internal.Column.*;
036import static org.elasticsearch.tools.opt.internal.Strings.*;
037
038/**
039 * <p>A means to display data in a text grid.</p>
040 *
041 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
042 */
043public class ColumnarData {
044    private static final String LINE_SEPARATOR = getProperty( "line.separator" );
045    private static final int TOTAL_WIDTH = 80;
046
047    private final ColumnWidthCalculator widthCalculator;
048    private final List<Column> columns;
049    private final String[] headers;
050
051    /**
052     * Creates a new grid with the given column headers.
053     *
054     * @param headers column headers
055     */
056    public ColumnarData( String... headers ) {
057        this.headers = headers.clone();
058        widthCalculator = new ColumnWidthCalculator();
059        columns = new LinkedList<Column>();
060
061        clear();
062    }
063
064    /**
065     * Adds a row to the grid.  The data will fall under the corresponding headers.
066     * There can be fewer elements in the row than headers.  Any data in columns outside
067     * of the number of headers will not be added to the grid.
068     *
069     * @param rowData row data to add
070     */
071    public void addRow( Object... rowData ) {
072        int[] numberOfCellsAddedAt = addRowCells( rowData );
073        addPaddingCells( numberOfCellsAddedAt );
074    }
075
076    /**
077     * Gives a string that represents the data formatted in columns.
078     *
079     * @return the formatted grid
080     */
081    public String format() {
082        StringBuilder buffer = new StringBuilder();
083
084        writeHeadersOn( buffer );
085        writeSeparatorsOn( buffer );
086        writeRowsOn( buffer );
087
088        return buffer.toString();
089    }
090
091    /**
092     * Removes all data from the grid, but preserves the headers.
093     */
094    public final void clear() {
095        columns.clear();
096
097        int desiredColumnWidth = widthCalculator.calculate( TOTAL_WIDTH, headers.length );
098        for ( String each : headers )
099            columns.add( new Column( each, desiredColumnWidth ) );
100    }
101
102    private void writeHeadersOn( StringBuilder buffer ) {
103        for ( Iterator<Column> iter = columns.iterator(); iter.hasNext(); )
104            iter.next().writeHeaderOn( buffer, iter.hasNext() );
105
106        buffer.append( LINE_SEPARATOR );
107    }
108
109    private void writeSeparatorsOn( StringBuilder buffer ) {
110        for ( Iterator<Column> iter = columns.iterator(); iter.hasNext(); )
111            iter.next().writeSeparatorOn( buffer, iter.hasNext() );
112
113        buffer.append( LINE_SEPARATOR );
114    }
115
116    private void writeRowsOn( StringBuilder buffer ) {
117        int maxHeight = max( columns, BY_HEIGHT ).height();
118
119        for ( int i = 0; i < maxHeight; ++i )
120            writeRowOn( buffer, i );
121    }
122
123    private void writeRowOn( StringBuilder buffer, int rowIndex ) {
124        for ( Iterator<Column> iter = columns.iterator(); iter.hasNext(); )
125            iter.next().writeCellOn( rowIndex, buffer, iter.hasNext() );
126
127        buffer.append( LINE_SEPARATOR );
128    }
129
130    private int arrayMax( int[] numbers ) {
131        int maximum = MIN_VALUE;
132
133        for ( int each : numbers )
134            maximum = Math.max( maximum, each );
135
136        return maximum;
137    }
138
139    private int[] addRowCells( Object... rowData ) {
140        int[] cellsAddedAt = new int[ rowData.length ];
141
142        Iterator<Column> iter = columns.iterator();
143        for ( int i = 0; iter.hasNext() && i < rowData.length; ++i )
144            cellsAddedAt[ i ] = iter.next().addCells( rowData[ i ] );
145
146        return cellsAddedAt;
147    }
148
149    private void addPaddingCells( int... numberOfCellsAddedAt ) {
150        int maxHeight = arrayMax( numberOfCellsAddedAt );
151
152        Iterator<Column> iter = columns.iterator();
153        for ( int i = 0; iter.hasNext() && i < numberOfCellsAddedAt.length; ++i )
154            addPaddingCellsForColumn( iter.next(), maxHeight, numberOfCellsAddedAt[ i ] );
155    }
156
157    private void addPaddingCellsForColumn( Column column, int maxHeight, int numberOfCellsAdded ) {
158        for ( int i = 0; i < maxHeight - numberOfCellsAdded; ++i )
159            column.addCell( EMPTY );
160    }
161}
