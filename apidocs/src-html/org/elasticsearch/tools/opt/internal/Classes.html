026package org.elasticsearch.tools.opt.internal;
027
028import java.util.HashMap;
029import java.util.Map;
030
031/**
032 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
033 */
034public final class Classes {
035    private static final Map<Class<?>, Class<?>> WRAPPERS = new HashMap<Class<?>, Class<?>>( 13 );
036
037    static {
038        WRAPPERS.put( boolean.class, Boolean.class );
039        WRAPPERS.put( byte.class, Byte.class );
040        WRAPPERS.put( char.class, Character.class );
041        WRAPPERS.put( double.class, Double.class );
042        WRAPPERS.put( float.class, Float.class );
043        WRAPPERS.put( int.class, Integer.class );
044        WRAPPERS.put( long.class, Long.class );
045        WRAPPERS.put( short.class, Short.class );
046        WRAPPERS.put( void.class, Void.class );
047    }
048
049    private Classes() {
050        throw new UnsupportedOperationException();
051    }
052
053    /**
054     * Gives the "short version" of the given class name.  Somewhat naive to inner classes.
055     *
056     * @param className class name to chew on
057     * @return the short name of the class
058     */
059    public static String shortNameOf( String className ) {
060        return className.substring( className.lastIndexOf( '.' ) + 1 );
061    }
062
063    /**
064     * Gives the primitive wrapper class for the given class. If the given class is not
065     * {@linkplain Class#isPrimitive() primitive}, returns the class itself.
066     *
067     * @param <T> generic class type
068     * @param clazz the class to check
069     * @return primitive wrapper type if {@code clazz} is primitive, otherwise {@code clazz}
070     */
071    public static <T> Class<T> wrapperOf( Class<T> clazz ) {
072        return clazz.isPrimitive() ? (Class<T>) WRAPPERS.get( clazz ) : clazz;
073    }
074}
