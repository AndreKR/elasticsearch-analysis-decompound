025package org.elasticsearch.tools.opt.util;
026
027import java.text.DateFormat;
028import java.text.ParsePosition;
029import java.text.SimpleDateFormat;
030import java.util.Date;
031
032import org.elasticsearch.tools.opt.ValueConversionException;
033import org.elasticsearch.tools.opt.ValueConverter;
034
035/**
036 * Converts values to {@link Date}s using a {@link DateFormat} object.
037 *
038 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
039 */
040public class DateConverter implements ValueConverter<Date> {
041    private final DateFormat formatter;
042
043    /**
044     * Creates a converter that uses the given date formatter/parser.
045     *
046     * @param formatter the formatter/parser to use
047     * @throws NullPointerException if {@code formatter} is {@code null}
048     */
049    public DateConverter( DateFormat formatter ) {
050        if ( formatter == null )
051            throw new NullPointerException( "illegal null formatter" );
052
053        this.formatter = formatter;
054    }
055
056    /**
057     * Creates a converter that uses a {@link SimpleDateFormat} with the given date/time pattern.  The date formatter
058     * created is not {@link SimpleDateFormat#setLenient(boolean) lenient}.
059     *
060     * @param pattern expected date/time pattern
061     * @return the new converter
062     * @throws NullPointerException if {@code pattern} is {@code null}
063     * @throws IllegalArgumentException if {@code pattern} is invalid
064     */
065    public static DateConverter datePattern( String pattern ) {
066        SimpleDateFormat formatter = new SimpleDateFormat( pattern );
067        formatter.setLenient( false );
068
069        return new DateConverter( formatter );
070    }
071
072    /** {@inheritDoc} */
073    public Date convert( String value ) {
074        ParsePosition position = new ParsePosition( 0 );
075
076        Date date = formatter.parse( value, position );
077        if ( position.getIndex() != value.length() )
078            throw new ValueConversionException( message( value ) );
079
080        return date;
081    }
082
083    /** {@inheritDoc} */
084    public Class<Date> valueType() {
085        return Date.class;
086    }
087
088    /** {@inheritDoc} */
089    public String valuePattern() {
090        return formatter instanceof SimpleDateFormat
091            ? ( (SimpleDateFormat) formatter ).toPattern()
092            : "";
093    }
094
095    private String message( String value ) {
096        String message = "Value [" + value + "] does not match date/time pattern";
097        if ( formatter instanceof SimpleDateFormat )
098            message += " [" + ( (SimpleDateFormat) formatter ).toPattern() + ']';
099
100        return message;
101    }
102}
