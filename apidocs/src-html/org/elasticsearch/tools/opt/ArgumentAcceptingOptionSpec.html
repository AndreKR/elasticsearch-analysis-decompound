026package org.elasticsearch.tools.opt;
027
028import java.util.ArrayList;
029import java.util.Collection;
030import static java.util.Collections.*;
031import java.util.List;
032import java.util.StringTokenizer;
033import static org.elasticsearch.tools.opt.internal.Objects.*;
034import static org.elasticsearch.tools.opt.internal.Reflection.*;
035import org.elasticsearch.tools.opt.internal.ReflectionException;
036import static org.elasticsearch.tools.opt.internal.Strings.*;
037
038/**
039 * <p>Specification of an option that accepts an argument.</p>
040 *
041 * <p>Instances are returned from {@link OptionSpecBuilder} methods to allow the formation of parser directives as
042 * sentences in a "fluent interface" language.  For example:</p>
043 *
044 * <pre>
045 *   <code>
046 *   OptionParser parser = new OptionParser();
047 *   parser.accepts( "c" ).withRequiredArg().<strong>ofType( Integer.class )</strong>;
048 *   </code>
049 * </pre>
050 *
051 * <p>If no methods are invoked on an instance of this class, then that instance's option will treat its argument as
052 * a {@link String}.</p>
053 *
054 * @param <V> represents the type of the arguments this option accepts
055 * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
056 */
057public abstract class ArgumentAcceptingOptionSpec<V> extends AbstractOptionSpec<V> {
058    private static final char NIL_VALUE_SEPARATOR = '\u0000';
059    
060    private boolean optionRequired;
061    private final boolean argumentRequired;
062    private ValueConverter<V> converter;
063    private String argumentDescription = "";
064    private String valueSeparator = String.valueOf( NIL_VALUE_SEPARATOR );
065    private final List<V> defaultValues = new ArrayList<V>();
066
067    ArgumentAcceptingOptionSpec( String option, boolean argumentRequired ) {
068        super( option );
069
070        this.argumentRequired = argumentRequired;
071    }
072
073    ArgumentAcceptingOptionSpec( Collection<String> options, boolean argumentRequired, String description ) {
074        super( options, description );
075
076        this.argumentRequired = argumentRequired;
077    }
078
079    /**
080     * <p>Specifies a type to which arguments of this spec's option are to be converted.</p>
081     *
082     * <p>JOpt Simple accepts types that have either:</p>
083     *
084     * <ol>
085     *   <li>a public static method called {@code valueOf} which accepts a single argument of type {@link String}
086     *   and whose return type is the same as the class on which the method is declared.  The {@code java.lang}
087     *   primitive wrapper classes have such methods.</li>
088     *
089     *   <li>a public constructor which accepts a single argument of type {@link String}.</li>
090     * </ol>
091     *
092     * <p>This class converts arguments using those methods in that order; that is, {@code valueOf} would be invoked
093     * before a one-{@link String}-arg constructor would.</p>
094     *
095     * <p>Invoking this method will trump any previous calls to this method or to
096     * {@link #withValuesConvertedBy(ValueConverter)}.
097     *
098     * @param <T> represents the runtime class of the desired option argument type
099     * @param argumentType desired type of arguments to this spec's option
100     * @return self, so that the caller can add clauses to the fluent interface sentence
101     * @throws NullPointerException if the type is {@code null}
102     * @throws IllegalArgumentException if the type does not have the standard conversion methods
103     */
104    public final <T> ArgumentAcceptingOptionSpec<T> ofType( Class<T> argumentType ) {
105        return withValuesConvertedBy( findConverter( argumentType ) );
106    }
107
108    /**
109     * <p>Specifies a converter to use to translate arguments of this spec's option into Java objects.  This is useful
110     * when converting to types that do not have the requisite factory method or constructor for
111     * {@link #ofType(Class)}.</p>
112     *
113     * <p>Invoking this method will trump any previous calls to this method or to {@link #ofType(Class)}.
114     *
115     * @param <T> represents the runtime class of the desired option argument type
116     * @param aConverter the converter to use
117     * @return self, so that the caller can add clauses to the fluent interface sentence
118     * @throws NullPointerException if the converter is {@code null}
119     */
120    @SuppressWarnings( "unchecked" )
121    public final <T> ArgumentAcceptingOptionSpec<T> withValuesConvertedBy( ValueConverter<T> aConverter ) {
122        if ( aConverter == null )
123            throw new NullPointerException( "illegal null converter" );
124
125        converter = (ValueConverter<V>) aConverter;
126        return (ArgumentAcceptingOptionSpec<T>) this;
127    }
128
129    /**
130     * <p>Specifies a description for the argument of the option that this spec represents.  This description is used
131     * when generating help information about the parser.</p>
132     *
133     * @param description describes the nature of the argument of this spec's option
134     * @return self, so that the caller can add clauses to the fluent interface sentence
135     */
136    public final ArgumentAcceptingOptionSpec<V> describedAs( String description ) {
137        argumentDescription = description;
138        return this;
139    }
140
141    /**
142     * <p>Specifies a value separator for the argument of the option that this spec represents.  This allows a single
143     * option argument to represent multiple values for the option.  For example:</p>
144     *
145     * <pre>
146     *   <code>
147     *   parser.accepts( "z" ).withRequiredArg()
148     *       .<strong>withValuesSeparatedBy( ',' )</strong>;
149     *   OptionSet options = parser.parse( new String[] { "-z", "foo,bar,baz", "-z",
150     *       "fizz", "-z", "buzz" } );
151     *   </code>
152     * </pre>
153     *
154     * <p>Then {@code options.valuesOf( "z" )} would yield the list {@code [foo, bar, baz, fizz, buzz]}.</p>
155     *
156     * <p>You cannot use Unicode U+0000 as the separator.</p>
157     *
158     * @param separator a character separator
159     * @return self, so that the caller can add clauses to the fluent interface sentence
160     * @throws IllegalArgumentException if the separator is Unicode U+0000
161     */
162    public final ArgumentAcceptingOptionSpec<V> withValuesSeparatedBy( char separator ) {
163        if ( separator == NIL_VALUE_SEPARATOR )
164            throw new IllegalArgumentException( "cannot use U+0000 as separator" );
165
166        valueSeparator = String.valueOf( separator );
167        return this;
168    }
169
170    /**
171     * Specifies a set of default values for the argument of the option that this spec represents.
172     *
173     * @param value the first in the set of default argument values for this spec's option
174     * @param values the (optional) remainder of the set of default argument values for this spec's option
175     * @return self, so that the caller can add clauses to the fluent interface sentence
176     * @throws NullPointerException if {@code value}, {@code values}, or any elements of {@code values} are
177     * {@code null}
178     */
179    public ArgumentAcceptingOptionSpec<V> defaultsTo( V value, V... values ) {
180        addDefaultValue( value );
181        defaultsTo( values );
182
183        return this;
184    }
185
186    /**
187     * Specifies a set of default values for the argument of the option that this spec represents.
188     *
189     * @param values the set of default argument values for this spec's option
190     * @return self, so that the caller can add clauses to the fluent interface sentence
191     * @throws NullPointerException if {@code values} or any elements of {@code values} are {@code null}
192     */
193    public ArgumentAcceptingOptionSpec<V> defaultsTo( V[] values ) {
194        for ( V each : values )
195            addDefaultValue( each );
196
197        return this;
198    }
199
200    /**
201     * Marks this option as required. An {@link OptionException} will be thrown when
202     * {@link OptionParser#parse(java.lang.String...)} is called, if an option is marked as required and not specified
203     * on the command line.
204     * 
205     * @return self, so that the caller can add clauses to the fluent interface sentence
206     */ 
207    public ArgumentAcceptingOptionSpec<V> required() {
208        optionRequired = true;
209        return this;
210    }
211    
212    public boolean isRequired() {
213        return optionRequired;
214    }
215
216    private void addDefaultValue( V value ) {
217        ensureNotNull( value );
218        defaultValues.add( value );
219    }
220
221    @Override
222    final void handleOption( OptionParser parser, ArgumentList arguments, OptionSet detectedOptions,
223        String detectedArgument ) {
224
225        if ( isNullOrEmpty( detectedArgument ) )
226            detectOptionArgument( parser, arguments, detectedOptions );
227        else
228            addArguments( detectedOptions, detectedArgument );
229    }
230
231    protected void addArguments( OptionSet detectedOptions, String detectedArgument ) {
232        StringTokenizer lexer = new StringTokenizer( detectedArgument, valueSeparator );
233        if ( !lexer.hasMoreTokens() )
234            detectedOptions.addWithArgument( this, detectedArgument );
235        else {
236            while ( lexer.hasMoreTokens() )
237                detectedOptions.addWithArgument( this, lexer.nextToken() );
238        }
239    }
240
241    protected abstract void detectOptionArgument( OptionParser parser, ArgumentList arguments,
242        OptionSet detectedOptions );
243
244    @SuppressWarnings( "unchecked" )
245    @Override
246    protected final V convert( String argument ) {
247        if ( converter == null )
248            return (V) argument;
249
250        try {
251            return converter.convert( argument );
252        }
253        catch ( ReflectionException ex ) {
254            throw new OptionArgumentConversionException( options(), argument, converter.valueType(), ex );
255        }
256        catch ( ValueConversionException ex ) {
257            throw new OptionArgumentConversionException( options(), argument, converter.valueType(), ex );
258        }
259    }
260
261    protected boolean canConvertArgument( String argument ) {
262        StringTokenizer lexer = new StringTokenizer( argument, valueSeparator );
263
264        try {
265            while ( lexer.hasMoreTokens() )
266                convert( lexer.nextToken() );
267            return true;
268        }
269        catch ( OptionException ignored ) {
270            return false;
271        }
272    }
273
274    protected boolean isArgumentOfNumberType() {
275        return converter != null && Number.class.isAssignableFrom( converter.valueType() );
276    }
277
278    public boolean acceptsArguments() {
279        return true;
280    }
281
282    public boolean requiresArgument() {
283        return argumentRequired;
284    }
285
286    public String argumentDescription() {
287        return argumentDescription;
288    }
289
290    public String argumentTypeIndicator() {
291        if ( converter == null )
292            return null;
293
294        String pattern = converter.valuePattern();
295        return pattern == null ? converter.valueType().getName() : pattern;
296    }
297
298    public List<V> defaultValues() {
299        return unmodifiableList( defaultValues );
300    }
301
302    @Override
303    public boolean equals( Object that ) {
304        if ( !super.equals( that ) )
305            return false;
306
307        ArgumentAcceptingOptionSpec<?> other = (ArgumentAcceptingOptionSpec<?>) that;
308        return requiresArgument() == other.requiresArgument();
309    }
310
311    @Override
312    public int hashCode() {
313        return super.hashCode() ^ ( argumentRequired ? 0 : 1 );
314    }
315}
