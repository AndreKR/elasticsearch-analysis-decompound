001    package org.xbib.elasticsearch.analysis.decompound;
002    
003    import java.io.IOException;
004    import java.io.InputStream;
005    
006    import org.elasticsearch.common.logging.ESLogger;
007    import org.elasticsearch.common.logging.Loggers;
008    import org.testng.annotations.Test;
009    import org.testng.Assert;
010    
011    public class DecompounderTests extends Assert {
012    
013        private final ESLogger logger = Loggers.getLogger(DecompounderTests.class.getName());
014    
015        @Test
016        public void testGerman() throws IOException, ClassNotFoundException {
017            InputStream forwfile = getClass().getResourceAsStream("/kompVVic.tree");
018            InputStream backfile = getClass().getResourceAsStream("/kompVHic.tree");
019            InputStream reducfile = getClass().getResourceAsStream("/grfExt.tree");
020            Decompounder d = new Decompounder(forwfile, backfile, reducfile, 0.51);
021            
022            String[] word = {
023                "Jahresfeier",
024                "Kinderernährung",
025                "Donaudampfschiff",
026                "Ökosteuer",
027                "Rechtsanwaltskanzleien",
028                "",
029                "tomaten"
030            };
031            String[] decomp = {
032                "[Jahr, feier]",
033                "[Kind, ernährung]",
034                "[Donau, dampf, schiff]",
035                "[Ökosteuer]",
036                "[Recht, anwalt, kanzlei]",
037                "[]",
038                "[tomaten]"
039            };
040        
041            for (int i = 0; i < word.length; i++) {
042                logger.info(word[i] +" => " + d.decompound(word[i]));
043                assertEquals(d.decompound(word[i]).toString(), decomp[i].toString());
044            }
045    
046        }
047    
048    }
