001    package org.xbib.elasticsearch.analysis.decompound;
002    
003    import java.io.BufferedReader;
004    import java.io.FileOutputStream;
005    import java.io.IOException;
006    import java.io.InputStream;
007    import java.io.InputStreamReader;
008    import java.io.Reader;
009    import java.util.zip.GZIPInputStream;
010    
011    public class LemmatizeBuildTests {
012    
013        public void trainReduce() throws IOException, ClassNotFoundException {
014            Trainer trainer = new Trainer();
015            CompactPatriciaTrie reduce = new CompactPatriciaTrie();
016            reduce.setIgnoreCase(true);
017            reduce.setReverse(true);
018            InputStream in = getClass().getResourceAsStream("/morphy-mapping-20110717.latin1.gz");
019            GZIPInputStream gzip = new GZIPInputStream(in);
020            Reader reader = new InputStreamReader(gzip, "ISO-8859-1");
021            BufferedReader br = new BufferedReader(reader);
022            String line;
023            while ((line = br.readLine()) != null) {
024                //System.err.println(line);
025                String[] forms = line.split("\t");
026                if (forms.length == 2) {
027                    String wclass = trainer.createRule(forms[0], forms[1]);
028                    //System.err.println("w=" + forms[0]  + " wclass="+ wclass);
029                    reduce.train(forms[0], wclass);
030                }
031            }
032            br.close();
033            FileOutputStream f = new FileOutputStream("target/morphyLemmaForms.tree");
034            reduce.save(f);
035            f.close();
036        }
037    }
