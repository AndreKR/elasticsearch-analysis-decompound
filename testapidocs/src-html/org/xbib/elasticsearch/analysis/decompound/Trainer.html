072    public void trainReduce(Reader reader) throws IOException, ClassNotFoundException {
073        reduce = new CompactPatriciaTrie();
074        reduce.setIgnoreCase(true);
075        reduce.setReverse(true);
076        BufferedReader br = new BufferedReader(reader);
077        String line;
078        while ((line = br.readLine()) != null) {
079            if (line.startsWith("#")) {
080                continue;
081            }
082            String[] forms = line.split(" ");
083            if (forms.length == 2) {
084                // <fullform> <baseform>
085                String wclass = createRule(forms[0], forms[1]);
086                //String cl = reduce.classify(line);
087                System.err.println("w=" + forms[0]  + " wclass="+ wclass);
088                reduce.train(forms[0], wclass);
089            }
090        }
091    }

210    public String createRule(String fullform, String baseform) {
211        String rule = "";
212        String part;
213        int bestlength = -1;
214        int start;
215        int end;
216        int occur1;
217        int occur2;
218        int length = fullform.length();
219        for (int i = 0; i < length; i++) {
220            for (int j = length; j > i; j--) {
221                part = fullform.substring(i, j);
222                occur1 = baseform.indexOf(part);
223                occur2 = fullform.indexOf(part);
224                if ((part.length() > bestlength) && occur1 == 0) {
225                    bestlength = part.length();
226                    start = i;
227                    end = j;
228                    rule = "";
229                    if (occur2 != 0) {
230                        rule = fullform.substring(0, start) + "#";
231                    }
232                    rule = rule + (fullform.length() - end) + baseform.substring(part.length(), baseform.length());
233                }
234            }
235        }
236        if (rule.equals("")) {
237            rule = fullform.length() + baseform;
238        }
239        return rule;
240    }

242    public String applyRule(String fullform, String wordClass) {
243        String baseform = "fehlende Behandlung in applyRule";
244        String pattern1 = "[0-9][0-9].*";
245        String pattern2 = "[0-9].*";
246        int temp;
247        if ("undecided".equals(wordClass)) {
248            baseform = "undecided";
249        } else if (wordClass.substring(0, 1).equals("‰")) {
250            temp = Integer.parseInt(wordClass.substring(1, 2));
251            baseform = fullform.substring(0, fullform.length() - temp);
252            baseform = substitute(baseform);
253        } else {
254            int i = wordClass.indexOf("#");
255            if (i != -1) {
256                baseform = fullform.substring(i);
257                if (wordClass.substring(i + 1).matches(pattern1)) {
258                    temp = Integer.parseInt(wordClass.substring(i + 1, i + 3));
259                    baseform = baseform.substring(0, baseform.length() - temp) + wordClass.substring(i + 3);
260                } else if (wordClass.substring(i + 1).matches(pattern2)) {
261                    temp = Integer.parseInt(wordClass.substring(i + 1, i + 2));
262                    baseform = baseform.substring(0, baseform.length() - temp) + wordClass.substring(i + 2);
263                }
264            } else {
265                if (wordClass.matches(pattern1)) {
266                    temp = Integer.parseInt(wordClass.substring(0, 2));
267                    baseform = fullform.substring(0, fullform.length() - temp) + wordClass.substring(2);
268                } else if (wordClass.substring(i + 1).matches(pattern2)) {
269                    temp = Integer.parseInt(wordClass.substring(0, 1));
270                    if (temp == 0) {
271                        baseform = fullform;
272                    } else {
273                        temp = Integer.parseInt(wordClass.substring(0, 1));
274                        baseform = fullform.length() > temp
275                                ? fullform.substring(0, fullform.length() - temp) + wordClass.substring(1)
276                                : fullform;
277                    }
278                }
279            }
280        }
281        return baseform;
282    }

284    private String substitute(String word) {
285        return word.replace('‰', 'a')
286                .replace('ƒ', 'A')
287                .replace('¸', 'u')
288                .replace('‹', 'U')
289                .replace('ˆ', 'o')
290                .replace('÷', 'O');
291    }
